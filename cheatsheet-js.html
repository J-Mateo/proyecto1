Variables

// var (EVITAR)
var nombre = "Juan";
var nombre = "Pedro";   // redeclarar ✅
nombre = "Carlos";      // reasignar ✅

// let (mutable, bloque)
let edad = 30;
edad = 35;

// const (fijo, referencia estable)
const PI = 3.1416;
// PI = 3.15; // ❌ No permitido
Tips
  • const por defecto, let si va a cambiar.
  • Evita var; su ámbito no es de bloque.

Operadores

// Aritméticos
5 + 3;
10 - 4;
3 * 4;
20 / 4;
10 % 3;
2 ** 3;

// Comparación estricta
5 === "5"; // false
5 !== "5"; // true

// Lógicos
true && false;
true || false;
!true;

// Ternario
const edad = 18;
const acceso = edad >= 18 ? "Permitido" : "Denegado";

// Nullish y optional chaining
const nombre = null;
console.log(nombre ?? "Anónimo");

const usuario = { direccion: { ciudad: "Madrid" } };
console.log(usuario.direccion?.ciudad);

Estructuras de control

// for
for (let i = 0; i < 3; i++) console.log(i);

// while
let c = 0;
while (c < 2) c++;

// if/else
const edad = 20;
if (edad >= 18) console.log("OK");
else console.log("NO");

// switch
switch ("Lunes") {
  case "Lunes": console.log("Inicio"); break;
  default: console.log("Normal");
}

Funciones

// Declarada
function saludar(nombre) {
  return \`Hola, ${nombre}\`;
}

// Expresada
const despedir = function(nombre) {
  return \`Adiós, ${nombre}\`;
};

// Arrow
const sumar = (a, b) => a + b;
const multiplicar = (a, b) => {
  const r = a * b;
  return r;
};

// Defaults y recursivas
function saludarPersona(nombre = "Anónimo") {
  console.log(\`Hola, ${nombre}\`);
}

function cuentaRegresiva(n) {
  if (n === 0) return console.log("BOOM");
  console.log(n);
  cuentaRegresiva(n - 1);
}

Métodos de Array

const a = [1,2,3,4,5];
const cuadrados = a.map(n => n ** 2);
const pares = a.filter(n => n % 2 === 0);
const suma = a.reduce((acc, n) => acc + n, 0);

a.find(n => n > 3);
a.includes(3);
a.concat([6,7]);
a.join(" - ");
a.forEach(n => console.log(n));
Consejo
  • Usa métodos inmutables (map, filter, reduce) siempre que sea posible.

Promesas

const promesa = new Promise((resolve, reject) => {
  const ok = true;
  ok ? resolve("✅ OK") : reject("❌ Error");
});

promesa
  .then(console.log)
  .catch(console.error)
  .finally(() => console.log("Fin"));

Fetch / Async-Await

// Fetch con then
fetch("[jsonplaceholder.typicode.com](https://jsonplaceholder.typicode.com/posts/1)")
  .then(res => res.json())
  .then(console.log)
  .catch(console.error);

// Async / await
async function cargar() {
  try {
    const res = await fetch("[pokeapi.co](https://pokeapi.co/api/v2/pokemon/1)");
    const data = await res.json();
    console.log(data);
  } catch(e) { console.error(e); }
}

Resumen

  • const por defecto, let si cambia.
  • Usa === y !== siempre.
  • Arrays inmutables con map/filter/reduce.
  • DOM: querySelector, classList, addEventListener.
  • Async: async/await + try/catch.
  • Storage: localStorage, sessionStorage y cookies.
